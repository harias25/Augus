print(2.1>2.3);         //False

print(1>=2.5);           //False
print(2.5>=2);           //True
print(2.1>=2.3);         //False
print("Hola">="Mundo");  //False

print("MENOR");

print(1<2.5);           //True
print(2.5<2);           //False
print(2.1<2.3);         //True
print("Hola"<"Mundo");  //True

print(1<=2.5);           //True
print(2.5<=2);           //False

print(2.5==2);           //False

print("Hola"=="Mundo");  //False
